
<html>
  <head>
    <meta charset="UTF-8" />
    <script
      type="text/javascript"
      src="http://ajax.googleapis.com/ajax/libs/jquery/1.6.2/jquery.min.js"
    ></script>

    <script type="text/javascript">
      function webServiceResult(data) {
        $("#resultado_json").empty();

        $("#resultado_json").append("oden_video: " + data.onden_video + "<br>");
      }

      // Llamada al webservice
      function callWebService() {
        try {
          $.ajax({
            url:
              "https://interfaz.cenart.gob.mx/api/video/catalogo/?id_post=28958",
            data: {
              q: "status"
            },
            type: "get",
            async: true,
            contentType: "text/json; charset=utf-8",
            dataType: "json",
            success: function (msg) {
              webServiceResult(msg);
            },
            error: function (jqXmlHttpRequest, textStatus, errorThrown) {
              alert("Error leyendo datos.");
            }
          });
        } catch (err) {
          alert(err);
        }
      }
    </script>
  </head>

  <body>
    <button type="button" onclick="javascript:callWebService();">
      Llamar a WS
    </button>

    <div id="resultado_json"></div>
  </body>
</html>
